# 34. 在排序数组中查找元素的第一个和最后一个位置

```c++
class Solution {
public:
    vector<int> searchRange(vector<int>& nums, int target) {
        int left = 0;
        int right = nums.size() - 1;
        vector<int> res;

        // find the left boundary
        while (left <= right) {
            int mid = left + (right - left) / 2;
            if (nums[mid] == target) {
                right = mid - 1;
            } else if (nums[mid] < target) {
                left = mid + 1;
            } else if (nums[mid] > target) {
                right = mid - 1;
            }
        }

        if (left == nums.size() || nums[left] != target) {
            vector<int> notFound;
            notFound.push_back(-1);
            notFound.push_back(-1);
            return notFound;
        }

        res.push_back(left);

        left = 0;
        right = nums.size() - 1;

        // find the right boundary
        while (left <= right) {
            int mid = left + (right - left) / 2;
            if (nums[mid] == target) {
                left = mid + 1;
            } else if (nums[mid] < target) {
                left = mid + 1;
            } else if (nums[mid] > target) {
                right = mid - 1;
            }
        }
        res.push_back(right);

        return res;
    }
};
```
